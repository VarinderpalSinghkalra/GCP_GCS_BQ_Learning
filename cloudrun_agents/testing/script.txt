#!/bin/bash

ORCHESTRATOR_URL="https://orchestrator-agent-platform-support-analyst-57565.us-central1.run.app"
INVENTORY_URL="https://inventory-agent-platform-support-analyst-57565.us-central1.run.app"
FINANCE_URL="https://finance-agent-platform-support-analyst-57565.us-central1.run.app"
LOGISTICS_URL="https://logistics-agent-platform-support-analyst-57565.us-central1.run.app"

echo "=============================="
echo "1. INVENTORY SEARCH"
echo "=============================="
curl "$INVENTORY_URL/inventory/search?name=bolt"
echo -e "\n"

echo "=============================="
echo "2. INVENTORY COST"
echo "=============================="
curl "$INVENTORY_URL/inventory/cost?item_id=REPLACE_UUID&quantity=10"
echo -e "\n"

echo "=============================="
echo "3. FINANCE CHECK"
echo "=============================="
curl -X POST "$FINANCE_URL/finance/check" \
  -H "Content-Type: application/json" \
  -d '{"amount":55}'
echo -e "\n"

echo "=============================="
echo "4. LOGISTICS EXECUTE"
echo "=============================="
curl -X POST "$LOGISTICS_URL/logistics/execute" \
  -H "Content-Type: application/json" \
  -d '{"order_id":"REPLACE_UUID","approved":true}'
echo -e "\n"

echo "=============================="
echo "5. ORCHESTRATOR (NO APPROVAL)"
echo "=============================="
curl -X POST "$ORCHESTRATOR_URL/query" \
  -H "Content-Type: application/json" \
  -d '{
    "item": "bolt",
    "quantity": 10
  }'
echo -e "\n"

echo "=============================="
echo "6. ORCHESTRATOR (WITH APPROVAL)"
echo "=============================="
curl -X POST "$ORCHESTRATOR_URL/query" \
  -H "Content-Type: application/json" \
  -d '{
    "item": "bolt",
    "quantity": 10,
    "approval": true
  }'
echo -e "\n"
